#include <stdio.h>

int main() {
    int a = 1;
    /* 포인터건 더블포인터건 트리플포인터건 그냥 주소저장함 */
    int* pa = &a;
    int** ppa = &pa;

    printf("%d \n", *pa);
    printf("%d \n", **ppa);
    
    return 0;
}

#include <stdio.h>

int main() {
    int a = 10;
    int b = 20;

    // int 값을 담는 배열이 아니라
    // int 변수들의 "주소"를 담는 배열
    int* arr[2];

    arr[0] = &a;
    arr[1] = &b;
    

    printf("%d\n", *arr[0]); // 10
    printf("%d\n", *arr[1]); // 20
    

    return 0;
}

#include <stdio.h>
#include <stdlib.h>

int main() {
    /* malloc : 실행중에 메모리를 자유자제로 할당해주기
    malloc 뜻: heap 공간에 공간 할당한다 */
    int ar[2];
    int ar2[] = { 1,2 };

    /* heap 영역에 공간 할당 하겠다. 정수 용도로
    heap영역에 공간 할당하고 거기 공간 주소가 H0 라고 가정하면
    main stack 변수 p에다가 H0 저장 하겠다 */

#include <stdio.h>
#include <stdlib.h>

int main() {
    int size = 3;

    int* arr = (int*)malloc(sizeof(int) * size);

    arr[0] = 10;
    arr[1] = 20;
    arr[2] = 30;

    printf("처음 배열:\n");
    for (int i = 0; i < size; i++) {
        printf("%d ", arr[i]);
    }

    // 배열 크기 늘리기
    size = 5;
    arr = (int*)realloc(arr, sizeof(int) * size);

    arr[3] = 40;
    arr[4] = 50;

    printf("\n\n크기 늘린 후:\n");
    for (int i = 0; i < size; i++) {
        printf("%d ", arr[i]);
    }

    free(arr);

    return 0;
}

#include <stdio.h>

void func(int** ar, int size) {
    for (int i = 0; i < size; i++) {
        *(*ar + i) = (*(*ar + i) + i);
    }
}

int main() {
    int ar[] = { 2,1, 3 };
    int* p = ar;
    int** pp = &p;
    int num = 4;

    func(pp, sizeof(ar) / sizeof(ar[0]); );
    num = ar[0];
    printf("%d", num);

    return 0;
}

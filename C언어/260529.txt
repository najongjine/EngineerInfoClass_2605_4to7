#include <stdio.h>

int main() {
    int a = 1;
    /* 포인터건 더블포인터건 트리플포인터건 그냥 주소저장함 */
    int* pa = &a;
    int** ppa = &pa;

    printf("%d \n", *pa);
    printf("%d \n", **ppa);
    
    return 0;
}

#include <stdio.h>

int main() {
    int a = 10;
    int b = 20;

    // int 값을 담는 배열이 아니라
    // int 변수들의 "주소"를 담는 배열
    int* arr[2];

    arr[0] = &a;
    arr[1] = &b;
    

    printf("%d\n", *arr[0]); // 10
    printf("%d\n", *arr[1]); // 20
    

    return 0;
}

#include <stdio.h>

/* 정렬 */
void printc1(char c1[],int size) {
	for (int a = 0; a < size;++a) {
		printf("%c ", c1[a]);
	}
	printf("\n");
}
int main() {
	char c1[4] = {
		'P','U','A','B'
	};
	int size = sizeof(c1) / sizeof(c1[0]);
	for (int a = 0; a < size-1; a+=1) {
		if (c1[a]>c1[a+1]) {
			char t = c1[a];
			c1[a] = c1[a + 1];
			c1[a + 1] = t;
		}
	}
	printc1(c1, size);
	return 0;
}

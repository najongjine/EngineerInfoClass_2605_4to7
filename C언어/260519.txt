/* 주석(메모장) 
*  주석에 쓴건 컴파일 안해요
* 
* #include 라는 뜻은 남이 만들어놓은
* 소스코드 복붙해라
* 어떤놈? stdio.h 에 들어있는 소스 코드들
* standard in out header
*/
#include <stdio.h>

/* void main () {}
* 메인함수
* 모든 프로그램은 시작시 무조건 main 함수부터 찾아요
*/
void main() {

	/* 이제부터 자유도가 생깁니다
	* main 함수 안에다가 우리가 하고싶은 연산을 적어요
	* 
	* 컴퓨터는
	* 산술연산(+ - * / %)
	* 비트연산( >> << & | ^ )
	* 논리연산( && || )
	* 제어연산( if, for )
	* 함수 function - 산술연산 조합해서 기계로 만들어 놓은것
	* 저장연산
	* 
	* 이걸 조합하면 GPT도 만들수 있다
	* 그러면 정보처리 코딩문제는 결국 연산종류 7개 해보기
	*/

	/* printf := 함수(기계)
	* 요구하는 입력이 있다
	* (문자열=R,데이터=Opt)
	* 흑백화면에 출력하는 기계에요
	* printf 라는걸 시험 문항에서 보면 무조건 종이에 써야된다
	* 
	* printf(1+1) 이걸 메모리에 올린다
	* (1+1 = 2) 가로안에 있는 연산을 먼저 한다
	* print(2) 화면에 보이는 기능을 수행 한다
	* 42번때 줄 니자고 나면 메모리에서 삭제 시킨다
	*/
	//printf("%d",1+1);
	/* 1+1 하라고 했어. 근데 저장하라고 했어?
	화면 화면에 보이라고 했어?
	컴퓨터는 생각보다 어마어마하게 멍청해요.
	그래서 더하기 하라고 했으면 지혼자 머릿속에서 더하기 해보고
	끝나요. 이걸 보여줄지, 저장할지도 사람이 일일히 명령 해줘야되요
	*/
	1 + 1;

	/*
	* 컴퓨터에게는 2가지의 데이터가 있습니다.
	* 상수데이터 : 숫자, 문자(a,b,c). 절대 변하지 않음
	* 변수데이터 : 변한다.
	*/

	/* 폰 노이만 모델은 메모리로 시작해서 메모리로 끝납니다
	* 정보처리기사 문제도 cpu 문제는 가뭄에 콩나듯나고
	* 죄다 메모리 문제 냅니다
	* 
	* 메모리는 바다같은 영역
	* 
	* Main stack, Heap stack, Static stack, 커스텀 stack
	* Main stack은 무조건 존재 해야합니다
	* 메모리 영역은 무조건 주소가 있습니다.
	* 하지만 주소는 매번 바뀝니다.
	* 
	* 변수라는건, 메모리 주소 어딘가에 별명을 붙이는 겁니다.
	*/
	int a;
	a = 1;
	/* 프로그래밍에선 = 요거는 같다라는 뜻이 아닙니다.
	프로그래밍에서 같다는 ==  요게 같다라는 뜻 
	= 이걸 저장연산 
	a라고 별명을 붙인 곳에다가 1을 저장해 */
	
	
	
	/* int := 자료형 data type 
	어떻게 생겨먹은 데이터니 
	int : 정수 (1,2,3,4, -1, -2, -3...) , 4byte
	long : 정수, 4byte
	short : 정수 2byte

	float : 실수(0.1, 1.1, 3.45...), 4byte
	double : 실수 , 8byte

	char : 문자 1byte
	*/
	int b = 1;
	int c=1;

	/* printf("%d %f %c %s \n",data1, data2, data3, data4)*/
	printf("나는 정수:%d \n", 11);
	printf("나는 실수:%f \n", 1.1);
	printf("나는 문자:%c \n", 'A');
	printf("나는 문자열: %s \n", "hello");


}

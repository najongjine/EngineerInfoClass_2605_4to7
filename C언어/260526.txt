#include <stdio.h>

int main() {
    int nums[3];
    char str[3];
    char c1 = 'A';
    /* 배열은 포인터에요. 배열은 heap 영역 어딘가의 시작주소를 가져요 */
    printf("nums가 가진거 뭐니 : %p \n", nums);
    printf("str아 넌 뭐 가졌니: %p \n", str);

    /* 얘는 포인터나 배열 아니에요. 그냥 상수 데이터 가져요 */
    printf("c1아 넌 뭐 가졌니: %c \n", c1);

    return 0;
}

#include <stdio.h>

int main() {
    int nums[3] = {1,2,3};
    /* 논리적으로 사람의 기준으로는 nums[1] 이래야 할거 같아요
    * 하지만 컴퓨터의 내부 동작 기준으로는 0부터 시작해야 되요
    * nums는 heap stack의 어딘가의 시작주소를 가지고 있다고 했어요.
    * 그런데 nums[1] 같이 사람기준으로 1을 넣어주면,
    * 얘가 가지고있는 (예: 0xH0 + 1) 이렇게 연산이 되버려서 1을 못가져오고
    * 2를 가져와 버려요.
    * 그래서 배열은 0부터 시작하는 거에요.
    * 배열의 속도는 O(1) 이에요. 1등이란 소리에요.
    * 메모리 어딘가의 시작주소 + 배열첨자번호 이렇게 하면
    * 크기가 천만인 배열도 속도가 드럽게 빨리 접근 되요.
    */
    printf("nums[0] == %d \n", nums[0]);
    

    return 0;
}

#include <stdio.h>

int main() {
    int nums[300] = {1,2,3};
    /* 반복문을 사용하면, 배열크기가 아무리 커져도 100% 다 접근가능해요*/
    for (int i = 0; i < 300;i++) {
        printf("nums[%d] == %d \n", i, nums[i]);
    }
    return 0;
}

#include <stdio.h>

int main() {
    /* 데이터 타입 나오고 [3] 요런거 나오면
    배열의 크기 정하기 */
    int nums[3] = {1,2,3};

    /* 첫번째 요소의 값 (0번째 인덱스의 값을) 5로 바꾸고 싶다 */
    nums[0] = 5;

    /* 데이터타입 없고 배열첨자 nums[0] 이런거 나오면,
    원산지 찾아가기 
    nums가 0xH0 가졌다 하면,
    0xH0 + 0 을 해서 원산지 찾아가고, nums[0] = 5 에서
    nums[0] 이게 = 사인 왼쪽에 있으니, 
    0xH0 + 0 공간에 5 라는 값을 저장해라.
    */

    nums[2] = 4;

    for (int i = 0; i < 3;i++) {
        printf("%d \n", nums[i]);
    }
    
    return 0;
}

#include <stdio.h>

int main() {
    /* 문자열의 비밀 */
    char c1 = 'H';
    char str1[1000] = "hello";
    printf("c1: %c \n", c1);
    printf("str1: %p \n", str1);
    for (int i = 0; str1[i] != '\0'; i++) {
        printf("%c", str1[i]);
    }

    /* %s := string의 약자.
    컴퓨터 입장에선 문자열은 긴 실과 같아요. 
    문자열 데이터 하나 나올때마다 반복문 쓰면 사람 죽어나니깐
    문자들이 heap 영역에 배치된건, %s, 변수명으로 편하게 쓸수있게 해줬다*/
    printf("\n c언어의 최초 편의성 패치: %s \n", str1);

    /* int 같은 숫자 데이터들을 %s로 출력하면 데이터 박살나요. */
    int nums[3] = { 3,2,'\0'};
    printf("\n nums: %s \n", nums);
    return 0;
}

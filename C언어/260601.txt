#include <stdio.h>
#include <string.h>

struct Student {
    char name[100];
    int score;
    char grade;
};
int main() {
    struct Student s1;
    strcpy_s(s1.name, sizeof(s1.name), "Minji");
    s1.score = 87;
    s1.grade = 'B';
    
    printf("name: %s \n score:%d \n grade:%c \n", 
        s1.name, s1.score, s1.grade);
    return 0;
}

#include <stdio.h>
#include <string.h>

typedef struct {
    char name[20];
    int score;
    char grade;
} Student;

int main(void) {
    Student list[3] = {
        {"Ara", 78, 'C'},
        {"Bora", 91, 'A'},
        {"Chan", 84, 'B'}
    };
    int i, sum = 0;

    for (i = 0; i < 3; i++) {
        sum += list[i].score;
        printf("%s:%d\n", list[i].name, list[i].score);
    }
    printf("avg=%d\n", sum / 3);
    return 0;
}

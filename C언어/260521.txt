#include <stdio.h>

int main() {
    /* 나이를 입력하라고 했는데 사람들이 장난 너무 많이침
    -100 누구는 9000 
    그래서 이거 막을거임 */
    int age = 90;

    if (age <= -1) { // 연인
        printf("장난치지 마세요");
    }
    else { // 사이
        printf("괜찬은가?");
    }
    if (age>=130) { // 솔로
        printf("장난치지 마세요");
    }
    
    
    
    return 0;
}

#include <stdio.h>

int main() {
    /* 나이를 입력하라고 했는데 사람들이 장난 너무 많이침
    -100 누구는 9000 
    그래서 이거 막을거임 */
    int age = -90;

    /*
    * 연산자 우선순위
    * () -> ! -> 비교연산 -> 논리연산
    * 가로가 있으니 가로안을 다 푸니 t 가 남았다
    * !t  이런게 남게되고, ! 표는 반전 이다.
    * t-> f, f->t 로 만든다
    */
    if (!(age >=0 && age <=130)) { 
        printf("장난치지 마세요");
    }
    
    
    
    return 0;
}

#include <stdio.h>

int main() {
    int n1 = 1; // 이건 main scope에 있는 n1
    if (n1) {
        /* 이건 main.if 에 scope 있는 n1 이다 */
        int n1 = 1; 
        n1++;
    }
    printf("%d \n", n1);
    
    
    return 0;
}

#include <stdio.h>

#include <stdio.h>

int main() {
    int n1 = 0;

    /* 후위연산은 나중에 한다.
    n1++ 은 없는셈 치고, 나중으로 미룸.
    n1 > 0    이거 검사함.
    검사 끝난후, n1 값 1 증가 */
    if (n1++ >0) {
        printf("실행함");

    }
    else {
        printf("실행안함");
    }
    
    return 0;
}

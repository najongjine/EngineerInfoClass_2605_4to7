#include <stdio.h>

int main() {
    int score = 60;
    if (score != 60) { // if(false)
        printf("같다!"); // false 면 안함
    }
    printf("나는 if문 다음줄에 있어요!"); // 이건 해야됨
    return 0;
}

    }
    else  { 
        /* else 는 if문이 false면 실행되는 놈이에요

#include <stdio.h>

int main() {
    /* 나이를 입력하라고 했는데 사람들이 장난 너무 많이침
    -100 누구는 9000 
    그래서 이거 막을거임 */
    int age = 90;

    if (age <= -1) { // 연인
        printf("장난치지 마세요");
    }
    else { // 사이
        printf("괜찬은가?");
    }
    if (age>=130) { // 솔로
        printf("장난치지 마세요");
    }
    
    
    
    return 0;
}



#include <stdio.h>

/* int := 리턴 타입.
얘가 퉤 뱉을때, 어떻게 생겨먹은 데이터 퉤 뱉어? */
int main() {
	float f1 = 1; // 실수를 저장
	char c1 = 'A';
	// %f 요놈은 내가 지정한 데이터 엮어서 보여야되
	printf("f1는 %.2f \n",f1);
	printf("c1: %c \n",c1);

	/* return main 함수 빠져 나오고, 0이라는 데이터 퉤 뱉어 */
	return 0;
}

#include <stdio.h>

int main() {
	int a = 1;
	int b = a + 2; // 3
	int c = a + b; // 4

	/* 이건 컴퓨터 입장에서 말이 되요.
	= 사인 기준으로 오른쪽은 값 복붙해오기, 왼쪽은 값 넣기 
	b의 값 복봍하고, a의 값 복봍해서 더한 결과를 b에 다가 다시 저장해요
	예전의 b값은 영영 사라져요 */
	b = b + a;
	return 0;
}

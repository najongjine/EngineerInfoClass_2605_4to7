

#include <stdio.h>

/* int := 리턴 타입.
얘가 퉤 뱉을때, 어떻게 생겨먹은 데이터 퉤 뱉어? */
int main() {
	float f1 = 1; // 실수를 저장
	char c1 = 'A';
	// %f 요놈은 내가 지정한 데이터 엮어서 보여야되
	printf("f1는 %.2f \n",f1);
	printf("c1: %c \n",c1);

	/* return main 함수 빠져 나오고, 0이라는 데이터 퉤 뱉어 */
	return 0;
}

#include <stdio.h>

int main() {
	int a = 1;
	int b = 2;
	/* 정수/정수 = 정수 */
	int c = a / b;
	printf("a(%d) / b(%d) = c(%d) \n", a, b, c);

	/* 나누기에서 소수점까지 나 나오게 고치기 */
	// 방법1 := 강제 캐스팅
	float c2 = (float)a / b;
	printf("a(%f) / b(%d) = c2(%f) \n", (float)a, b, c2);

	// 방법2 := float 으로 타입 선언하기

	float a2 = 1;
	c2 = a2 / b;
	printf("a2(%f) / b(%d) = c2(%f) \n", a2, b, c2);

	/* 핵심은 c언어랑 JAVA는 나누기할때, 데이터중 단 하나라도
	소수점이 들어가면 소수점을 남긴다.
	하지만 나누는 데이터가 모두 정수면, 결과는 정수로 나온다 */


	return 0;
}

#include <stdio.h>

int main() {
	int a = 1;
	/* ++ , -- 이걸 단항연산자 라고 해요
	다른 데이터와 결합 없이,   = 도 없이 데이터 변경이 가능 해요 */
	
	/* 단항증가, 단항감소 연산자는 2가지 종류가 있어요
	하나는 전위연산, 다른하나는 후위연산 이라고 해요 */
	
	printf("%d \n",a++);


	/* 전위연산 : 딴게 어쩌구저쩌구 막 있어도, 값 증가부터해
	곱하기나 가로 먼저 하는게 아니라, a 메모리 영역 찾아가서
	++ 먼저 하라고 */
	int b = ++a + (1 * 2);
	
	a = 1; b = 0;
	/* 후위연산: 가로쳐지고 뭐 막 있어도, 나중에 하라고  */
	b = (a++) + 1 * 2; // a++ 보류 
	printf("b: %d \n", b);
	printf("a: %d \n", a);


	return 0;
}

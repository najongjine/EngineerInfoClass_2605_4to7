

#include <stdio.h>

/* int := 리턴 타입.
얘가 퉤 뱉을때, 어떻게 생겨먹은 데이터 퉤 뱉어? */
int main() {
	float f1 = 1; // 실수를 저장
	char c1 = 'A';
	// %f 요놈은 내가 지정한 데이터 엮어서 보여야되
	printf("f1는 %.2f \n",f1);
	printf("c1: %c \n",c1);

	/* return main 함수 빠져 나오고, 0이라는 데이터 퉤 뱉어 */
	return 0;
}

#include <stdio.h>

int main() {
	int a = 1;
	int b = a + 2; // 3
	int c = a + b; // 4

	/* 이건 컴퓨터 입장에서 말이 되요.
	= 사인 기준으로 오른쪽은 값 복붙해오기, 왼쪽은 값 넣기 
	b의 값 복봍하고, a의 값 복봍해서 더한 결과를 b에 다가 다시 저장해요
	예전의 b값은 영영 사라져요 */
	b = b + a;
	return 0;
}

#include <stdio.h>

int main() {
	int a = 1;
	int b = 2;
	/* 정수/정수 = 정수 */
	int c = a / b;
	printf("a(%d) / b(%d) = c(%d) \n", a, b, c);

	/* 나누기에서 소수점까지 나 나오게 고치기 */
	// 방법1 := 강제 캐스팅
	float c2 = (float)a / b;

	// 방법2 := float 으로 타입 선언하기

	float a2 = 1;
	c2 = a2 / b;

	/* 핵심은 c언어랑 JAVA는 나누기할때, 데이터중 단 하나라도
	소수점이 들어가면 소수점을 남긴다.
	하지만 나누는 데이터가 모두 정수면, 결과는 정수로 나온다 */


	return 0;
}

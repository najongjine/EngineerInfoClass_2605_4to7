#include <stdio.h>

int main() {
    int i = 9999990;
    /*
    * 첫구간 : 변수값 초기화 구간. forloop 만났을때 단 한번만 수행
    * 두번째구간: 조건검사 구간. true면 반복문 블록안에 있느거 수행
    * 3번째: 연산구간. 변수값 변경
    */
    for (i = 0; 5 ;i+=2) {
        printf("hello \n");
    }

    return 0;
}

#include <stdio.h>

int main() {
    int i = 0;
    int sum = 0;

    /* forloop 에서 변수 선언하면 되게 조심해야 되요
    매 회차때마다(forloop 안에있는 연산식 수행후)
    block 안에있는 변수들 싹 지워요. 그리고 다시시작해요
    */
    for (; i< 5 ;i++) {
        int sum = 0;
        sum += i;
        printf("sum:%d \n", sum);
    }
   
    printf("최후의 i값: %d \n", i);
    return 0;
}

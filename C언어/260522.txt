#include <stdio.h>

int main() {
    int i = 9999990;
    /*
    * 첫구간 : 변수값 초기화 구간. forloop 만났을때 단 한번만 수행
    * 두번째구간: 조건검사 구간. true면 반복문 블록안에 있느거 수행
    * 3번째: 연산구간. 변수값 변경
    */
    for (i = 0; 5 ;i+=2) {
        printf("hello \n");
    }

    return 0;
}

#include <stdio.h>

int main() {
    int i = 0;

    for (; i< 5 ;i++) {
        int sum = 0;
        sum += i;
        printf("sum:%d \n", sum);
    }
   
    printf("최후의 i값: %d \n", i);
    return 0;
}

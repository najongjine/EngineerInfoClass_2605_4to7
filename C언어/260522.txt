#include <stdio.h>

int main() {
    int i = 9999990;
    /*
    * 첫구간 : 변수값 초기화 구간. forloop 만났을때 단 한번만 수행
    * 두번째구간: 조건검사 구간. true면 반복문 블록안에 있느거 수행
    * 3번째: 연산구간. 변수값 변경
    */
    for (i = 0; 5 ;i+=2) {
        printf("hello \n");
    }

    return 0;
}

#include <stdio.h>

int main() {
    int i = 0;
    int sum = 0;

    /* forloop 에서 변수 선언하면 되게 조심해야 되요
    매 회차때마다(forloop 안에있는 연산식 수행후)
    block 안에있는 변수들 싹 지워요. 그리고 다시시작해요
    */
    for (; i< 5 ;i++) {
        int sum = 0;
        sum += i;
        printf("sum:%d \n", sum);
    }
   
    printf("최후의 i값: %d \n", i);
    return 0;
}

#include <stdio.h>

int main() {
    /* 반복문이 중첩이 되있을땐 굉장히 어려워요
    * 노가다 양이 기하급수적이 되거든요.
    * 자식 forloop를 만나면, 그놈이 다 끝나기 전까진 부모
    * forloop의 i++을 가지 않아요.
    * 부모 loop의 ++ 을 하고나면 다시 자식 forloop 만나는데,
    * 자식은 또 0부터 시작해요
    */
    for (int a = 0; a < 3;a++) {
        printf("a:%d \n",a);
        for (int b = 0; b < 2;b++) {
            printf("b:%d \n", b);
        }
    }
    
    return 0;
}

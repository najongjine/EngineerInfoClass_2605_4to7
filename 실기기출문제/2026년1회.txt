1. 다음은 C언어에 대한 문제이다. 아래 코드를 확인하여 알맞는 출력값을 작성하시오.
#include <stdio.h>
// 배열이나 포인터랑 같아요
double arr1(int p[], int len) {
    double av = 0; // 530
    int i;
    // 0~9  av에다가 배열에 있는거 싹 더하래요
    for (i = 0; i < len; i++) {
        av += (double)p[i];
    }
    // 530/10 =53
    return av / len;
}
// 배열이나 포인터랑 같아요. arr1 랑 같은 코드에요
double arr2(int* p, int len) {
    double av = 0;
    int i;
    for (i = 0; i < len; i++) {
        av += (double)(*(p + i));
    }
    // 530/10 =53
    return av / len;
}

int main() {
    int arr[10] = {
        80,
        20,
        50,
        55,
        45,
        95,
        55,
        10,
        40,
        80
    };
    int len = 10;
    //                   53      +   53 = 106
    // %.2f == 소수점 2자리까지 써라 == 106.00
    printf("%.2f", arr1(arr, len) + arr2(arr, len));

    return 0;
}

2. 다음 설명에 해당하는 디자인 패턴의 유형을 괄호( ) 안에 쓰시오.
Bridge
 - 추상부와 구현부를 나누어 개발하고 연결
 - 기능의 클래스 계층과 구현의 클래스 계층을 연결
Observer
 - 객체의 상태가 바뀌면 그걸 구독/의존 하는놈들이 알아서 작동한다
 - 한 객체의 상태가 바뀌면 그 객체에 의존

1. 다음은 C언어에 대한 문제이다. 아래 코드를 확인하여 알맞는 출력값을 작성하시오.
#include <stdio.h>
// 배열이나 포인터랑 같아요
double arr1(int p[], int len) {
    double av = 0; // 530
    int i;
    // 0~9  av에다가 배열에 있는거 싹 더하래요
    for (i = 0; i < len; i++) {
        av += (double)p[i];
    }
    // 530/10 =53
    return av / len;
}
// 배열이나 포인터랑 같아요. arr1 랑 같은 코드에요
double arr2(int* p, int len) {
    double av = 0;
    int i;
    for (i = 0; i < len; i++) {
        av += (double)(*(p + i));
    }
    // 530/10 =53
    return av / len;
}

int main() {
    int arr[10] = {
        80,
        20,
        50,
        55,
        45,
        95,
        55,
        10,
        40,
        80
    };
    int len = 10;
    //                   53      +   53 = 106
    // %.2f == 소수점 2자리까지 써라 == 106.00
    printf("%.2f", arr1(arr, len) + arr2(arr, len));

    return 0;
}

14. 아래 파이썬 코드를 실행했을 때 출력되는 값을 쓰시오.
def f(a):
    """
    [[1], [2], [3], [4]]
    """
    m = [[x] for x in a]
    """
    주소 공유가 아니라, 내용물 자체를 복사해서,
    새로운 배열 만들기
    """
    b = m[:]
    #   i=0~2            3
    for i in range(len(b) - 1):
        """
        b[0+1]+=b[0]
        b[1]= b[1] + b[0]
        b[1]= [2] + [1]
        b[1] = [2,1]
        [[1], [2,1], [3], [4]]

        b[1+1]+=b[1]
        b[2]= b[2] + b[1]
        b[2] = [3] + [2,1]
        b[2] = [3,2,1]
        [[1], [2,1], [3,2,1], [4]]
        ...
        [[1], [2,1], [3,2,1], [4,3,2,1]]
        """
        b[i+1] += b[i]
    """
    list comprehension
    [1] -> x 에 가고, len([1]) -> 1
    [2,1] -> x 에 가고, len([2,1]) -> 2
    ...3
    ...4
    """
    return sum(len(x) for x in m)
#f([1, 2, 3, 4])
print(f([1, 2, 3, 4]))

14. 다음 코드에서 괄호안에 알맞는 값을 변수명으로 작성하시오.

package dummy;

public class Test11 {	
	public static void swap(int[] arr, int idx1, int idx2){
        /*
        프로그래밍에서 자리를 바꿀땐 의자 3개가 필요하다(python 제외)
        x 자 cross 체크
        */
        int temp = arr[idx1];
        arr[idx1] = arr[idx2];
        // 여기에 변수명
        arr[idx2] = temp;
    }
 
    public static void Usort(int[] array, int length){
        for (int i = 0; i < length; i++) {
            for (int j = 0; j < length - i - 1; j++) {
                /*
                왼쪽에 있는게 오른쪽에 있는것보다 크니?
                그러면 자리 바꿔
                */
                if (array[j] > array[j + 1]) {
                    swap(array, j, j + 1);
                }
            }
        }
    }
    
	public static void main(String[] args) {
		int[] item = new int[] { 5, 3, 8, 1, 2, 7 };
        int nx = 6;   
        // 여기에 변수명
        Usort(item, nx);
 
        for (int data : item) {
            System.out.print(data + " ");
        }
	}
}

package dummy;

class Parent {
    int x = 100; // 500
 
    Parent() {
        this(500);
    }
 
    Parent(int x) {
        this.x = x;
    }
 
    int getX() {
    	// 이쪽 method의 scope에서 가장 가까운 x는 Parent에 있는 field x
    	// 그래서 100이 return 됨
        return x;
    }
}
class Child extends Parent {
    int x = 4000; // 5000
    // 여기로 먼저 진입
    // 부모 생성자 호출하는거 잊지말기
    Child() {
    	// super(); 이거 부모생성자 호출 안적어도 컴파일때 내부적으로 지가 만듬
        this(5000); // 자시자신 생성자 호출
    }
    // 그래서 여기도 실행
    Child(int x) {
        this.x = x;
    }
}
public class Test11 {	
	public static void main(String[] args) {
		 Child obj = new Child();
	     System.out.println(obj.getX());
	}

1. 아래 자바 코드에서 출력되는 값을 작성하시오.

class Static{
 
    public int a = 20;
    // static은 static 영역에 딱 하나만 존재하고, 모든 곳에서 접근 가능
    static int b = 0;
    
}
 
public class Main{
    public static void main(String[] args) {
        
        int a;
        a = 10;
        Static.b = a;

        Static st = new Static();
        // 후위 연산자는 뭐가 있건간에 나중에 ++을 한다. 출력시에는 10으로 출력.
        System.out.println(Static.b++);
        // st.b 도 결국은 Static 영역에서 b 가져옴
        System.out.println(st.b);
        System.out.println(a);
        System.out.print(st.a);
    }

}

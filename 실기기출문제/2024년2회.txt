1. 다음은  Java 코드에 대한 문제이다. 아래 코드를 확인하여 알맞는 출력값을 작성하시오.
package dummy;

public class Test11 {	
	/* 배열들이 모두 new로 새로운 공간에 할당됬기 때문에 주소값은 절대 같을수 없다 */
	public static void check(int[] a, int[] b) {
        if (a==b) {
            System.out.print("O");
        }else{
            System.out.print("N");
        }
    }
	public static void main(String[] args) {
		// heap 영역 새로운 주소 할둥후 a에 주소저장
		int[] a = new int[]{1, 2, 3, 4};
		// heap 영역 새로운 주소 할둥후 b에 주소저장
        int[] b = new int[]{1, 2, 3, 4};
		// heap 영역 새로운 주소 할둥후 c에 주소저장
        int[] c = new int[]{1, 2, 3};
        
        check(a, b);
        check(a, c); 
        check(b, c); 
	}
}

# x=abdcabcabca   y="ab"
def fnCalculation(x,y):
    result = 0; # 1
    # 0 ~ 10   
    for i in range(len(x)): # 11    i=2
     #      x[2:4] == dc
     # temp=dc
     temp = x[i:i+len(y)] 
     #  dc == ab
     if temp == y:
       result += 1;
    return result
 
a = "abdcabcabca"
p1 = "ab";
p2 = "ca";
#               3                    3
out = f"ab{fnCalculation(a,p1)}ca{fnCalculation(a,p2)}"

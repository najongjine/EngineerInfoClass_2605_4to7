package dummy;

class A {
    // 초기는 부모로 오고 override 체크
    public void paint() {
        System.out.print("A");
        draw();
    }
    public void draw() {
        System.out.print("B");
        // override 
        draw();
    }
}
 
class B extends A {
    // 자식에서 override 되있으니 이거 실행
    public void paint() {
        // 부모꺼 강제호출
        super.draw();
        System.out.print("C");
        // 자기꺼 호출
        this.draw();
    }
    public void draw() {
        System.out.print("D");
    }
}
public class Test11 {	
	public static void main(String[] args) {
        // upcasting. 포커스는 A로 맞춤
		A b = new B();
        b.paint();
        b.draw();
	}
}

10. 다음은 C언어의 포인터 문제이다. 알맞는 출력값을 작성하시오.
#include <stdio.h>
#include <stdlib.h>
int main() {
    // p 는 H0를 저장하고 있다고 가정
    char* p = "KOREA";
    //%s는 편의성 기능이 들어가 있어서 *p 이렇게 안해도 문자열 내용 다 나와요
    printf("%s\n", p); // KOREA

    //            H0 + 1 = H1. 시작위치를 H1으로 잡고 문자열 출력해
    printf("%s\n", p + 1); // OREA

    // %c는 문자에요. %s같은 편의성 기능 없어요
    printf("%c\n", *p); // K

    //             *(H0 + 3) 
    printf("%c\n", *(p + 3)); // E

    //             'K' + 4    (쉽게 알파벳 K 에서 4번정도 가면 O 나오는걸로 계산)
    printf("%c\n", *p + 4); // O
}

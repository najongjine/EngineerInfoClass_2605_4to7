package dummy;

9. 다음은 C언어 문제이다. 알맞은 출력값을 작성하시오.

#include <stdio.h>
#define MAX_SIZE 10

/* Stack 자료구조 알고리즘
* 배열을 추적하는 index 변수 하나를 가지고 배열이 다 찼나, 배열이 아예 비어있나 다 검사함
* 값을 넣거나 빼는곳도 index 변수 하나의 위치로 결정남
*/
int isWhat[MAX_SIZE];
int point = -1;

// 배열이 완전 비어있니 체크
int isEmpty() {
    if (point == -1) return 1; // true
    return 0; // false
}

// 배열이 꽉 찼니 체크
int isFull() {
    if (point == 10) return 1;
    return 0;
}

// 배열에 값 넣기
void into(int num) {
    if (point >= 10) printf("Full");
    else isWhat[++point] = num;
}

// 배열에서 값 가져오기
int take() {
    if (isEmpty() == 1) printf("Empty");
    else return isWhat[point--];
    return 0;
}

int main(int argc, char const* argv[]) {
    int e;
    into(5); into(2);

    // ! 이건 0을 1로, 1을 0으로 반전 시키기
    while (!isEmpty()) {
        printf("%d", take());
        into(4); into(1); printf("%d", take());
        into(3); printf("%d", take()); printf("%d", take());
        into(6); printf("%d", take()); printf("%d", take());
    }

    return 0;
}

public class Main{
    public static void main(String[] args) {

    }

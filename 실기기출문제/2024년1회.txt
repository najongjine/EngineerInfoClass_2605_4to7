package dummy;

class Connection {
	/* static 공간은 Heap 영역 어딘가에 할당된 시작주소를 평생 가지고 있는다
	 * 모든 영역에서 그 공간 참조 가능하게 됬다 */
    private static Connection _inst = null;
    private int count = 0;
    
    static public Connection get() {
        if(_inst == null) {
        	/* 메모리 공간 어딘가에 공간 만들고, 필요한 데이터랑 함수 셋팅하고
        	 * 그곳의 시작 수조를 넘겨준다 */
            _inst = new Connection();
            return _inst;
        }
        /* 전에 공간 만들어둔거 있으면 그거 계속 재활용해
         * 그게 싱글톤 패턴이다 */
        return _inst;
    }
    
    public void count() {
         count++; 
    }
    
    public int getCount() {
         return count; 
    }
}
public class Test11 {	
	public static void main(String[] args) {
		/* 아무리 Connection.get()을 많이 호출해도
		 * 하나 만들어둔 공간 계속 똑같이 배정 받게 된다 */
		 Connection conn1 = Connection.get();
		 /* H0라는(가정) 공간에 있는 count 변수값 증가시켜라*/
	        conn1.count();
	 
	        Connection conn2 = Connection.get();
	        /* H0라는(가정) 공간에 있는 count 변수값 증가시켜라*/
	        conn2.count();
	 
	        Connection conn3 = Connection.get();
	        /* H0라는(가정) 공간에 있는 count 변수값 증가시켜라*/
	        conn3.count();
	        /* H0라는(가정) 공간에 있는 count 변수값 증가시켜라*/
	        conn1.count();
	        System.out.print(conn1.getCount()); // 4

#include <stdio.h>
#include <string.h>
void reverse(char* str) {
    // strlen == 문자열의 기능을 자동으로 계산해주는 모듈
    int len = strlen(str);
    printf("len : %d \n", len);
    char temp;
    char* p1 = str;
    char* p2 = str + len - 1;
    while (p1 < p2) {
        temp = *p1;
        *p1 = *p2;
        *p2 = temp;
        p1++;
        p2--;
    }
}

    char str[100] = "ABCDEFGH";

    reverse(str);

    int len = strlen(str); // 8.   \0 count 안함

    for (int i = 1; i < len; i += 2) {
        printf("%c", str[i]);
    }

    printf("\n");

11. 다음 C언어의 알맞는 출력값을 작성하시오.
#include <stdio.h>
#include <string.h>

typedef struct {
    int accNum;
    double bal;
}BankAcc;
//
double sim_pow(double base, int year) {
    int i;
    double r = 1.0;

    for (i = 0; i < year; i++) {
        r = r * base;
    }
    return r;
}

void initAcc(BankAcc* acc, int x, double y) {
    acc->accNum = x;
    acc->bal = y;
}

void xxx(BankAcc* acc, double* en) {
    if (*en > 0 && *en < acc->bal) {
        acc->bal = acc->bal - *en;
    }
    else {
        acc->bal = acc->bal + *en;
    }
}

void yyy(BankAcc* acc) {
    //            2100 * 1.331 = 2,795.1
    acc->bal = acc->bal * sim_pow((1 + 0.1), 3);
}
int main() {

    BankAcc myAcc;
    initAcc(&myAcc, 9981, 2200.0);
    double amount = 100.0;
    xxx(&myAcc, &amount);
    yyy(&myAcc);
    printf("%d and %.2f", myAcc.accNum, myAcc.bal);
    return 0;

}

package dummy;
class classOne {
    int a, b; // 10, 11
    //                 10  11
    public classOne(int a, int b) {
        this.a = a;
        this.b = b;
    }
    // 일단 여기로 오고나서 override 확인
    public void print() {
        System.out.println(a + b);
    }
 
}
class classTwo extends classOne {
    int po = 3;
    //                 10
    public classTwo(int i) {
    	//    10,  11
        super(i, i+1);
    }
    // override 있으니깐 이거 실행
    public void print() {
    	// 이 method scope에서 제일 가까운 po 변수는 classTwo 에 있는 po 필드
        System.out.println(po*po);
    }
}
public class Test11 {	
	public static void main(String[] args) {
		classOne one = new classTwo(10);
        one.print();
	}
}

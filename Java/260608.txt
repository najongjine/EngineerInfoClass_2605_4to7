package itinfo_2605_4to7;

import java.util.ArrayList;
import java.util.List;
import java.util.Random;

/* 추상클래스를 굳이 왜 만들었냐?
 * 상속을 많이 쓰게 하려고
 * 왜 상속을 만들었냐?
 * 공통된 속성이나 함수 여기다가 적어서 중복같은거 줄이려고 */
abstract class Animal {
	int a=1;
     void sound() {
    	 System.out.println("크르릉");
     };
     /* abstract function 
      * - 만들다 만 함수처럼 생김.
      * 자식쪽에서 override 해서 구채화 해줘야함 */
     abstract void attack();
}
/* 자식은 상속 받을때랑 똑같이 작동해요 */
class Dog extends Animal{
	@Override
	void attack() {
		System.out.println("물기");
		
	}
}

public class Test1 {
	public static void main(String[] args) {
		Dog a=new Dog();
      System.out.println(a.a);
      a.sound();
	}
}

package itinfo_2605_4to7;

import java.util.ArrayList;
import java.util.List;
import java.util.Random;

abstract class Animal {
	 int eye=2;
	 int nose=1;
	 int leg=4;
	 Animal(){
		 System.out.println("동물탄생");
	 }
     void sound() {
    	 System.out.println("크르릉");
     };

     abstract void eat();
}
/* 자식은 상속 받을때랑 똑같이 작동해요 */
class Dog extends Animal{
	Dog(){
		System.out.println("개 탄생");
	}
	void sound() {
		System.out.println("멍멍");
	}
	void eat() {
		System.out.println("큰거도 이빨로 씹어 먹어요");
	}
}
class Cat extends Animal{
	void sound() {
		System.out.println("야옹");
	}
	void eat() {
		System.out.println("작은음식 이빨로 씹어 먹어요");
	}
}
class Snake extends Animal{
	int leg=0;
	int nose=0;
	void sound() {
		System.out.println("...");
	}
	void eat() {
		System.out.println("삼켜요");
	}
}

public class Test1 {
	public static void main(String[] args) {
		/* 다형성을 적용하면
		 * new는 다르게 한 놈들이라도 타입이 같으니
		 * list에 넣을수 있다. */
		List<Animal> alist=new ArrayList<>();
		
		Animal a1=new Dog();
		Animal a2=new Cat();
		Animal a3=new Snake();
		alist.add(a3);
		/* 다형성
		 * - 사용하는 코드는 같지만, 결과나 기능은 다 다르게 작동함 */
		a1.sound();
		a2.sound();
		a3.sound();
	}
}

package itinfo_2605_4to7;

import java.util.ArrayList;
import java.util.List;
import java.util.Random;
class Device {
    String name;

    Device(String name) {
        this.name = name;
    }

    void info() {
        System.out.println(name);
    }
}
interface Chargeable {
    void charge();
}
class Phone extends Device implements Chargeable {
    Phone(String name) {
        super(name);
    }

    public void charge() {
        System.out.println(name + ":charge");
    }
}

public class Test1 {
	public static void main(String[] args) {
		Phone p = new Phone("Phone");
        p.info();
        p.charge();
	}
}

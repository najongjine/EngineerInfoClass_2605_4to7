package itinfo_2605_4to7;

import java.util.ArrayList;
import java.util.List;
import java.util.Random;

/* 사용자가 만든 자료형
 * 꼭 단어를 쓰는것같이 가독성이 높아지고, 또 정한값만 가질수 있어서
 * 통일성이 있음 */
enum OrderStatus{
	READY, SHIPPED, CANCLE
}
public class Test1 {
	public static void main(String[] args) {
		// 단순 숫자로 상태를 표현하면 햇갈림
		 int nstatus=3;
		 // enum으로 상태를 표현하면 가동성이 좋음
		 OrderStatus status=OrderStatus.READY;
		 if(status==OrderStatus.READY) {
			 System.out.println("주문 완료");
		 }
		 else if(status==OrderStatus.SHIPPED) {
			 System.out.println("배송 완료");
		 }
		 if(nstatus==0) {
			 System.out.println("준비중");
		 }
		 if(nstatus==5) {
			 System.out.println("취소");
		 }
	}
}

package itinfo_2605_4to7;

import java.util.ArrayList;
import java.util.List;
import java.util.Random;

/* 이 예제는 ENUM을 이악물고 숫자로 보고싶다 할때 쓰는 방법 */
enum Level {
    LOW(1), // ENUM값에 숫자값을 정해줌
    MIDDLE(2),
    HIGH(3);

    int score; // 숫자를 담을 변수

 // ENUM 옆에 있는 값을 스스로 가져가서 score에 셋팅하는 생성자
    Level(int score) { 
        this.score = score;
    }
}
public class Test1 {
	public static void main(String[] args) {
		/* 그럼 사용할때 자동으로 생성자에 숫자 넘겨서 score에 셋팅함 */
		System.out.println(Level.LOW.score);
        System.out.println(Level.HIGH.score);
	}
}

package itinfo_2605_4to7;

import java.util.ArrayList;
import java.util.List;
import java.util.Random;

enum Size {
    SMALL(10),
    LARGE(20);

	// 접근자 private을 붙이면 외부에서(main, 타 클래스든) 접근이 안되요
    private int price;

    Size(int price) {
        this.price = price;
    }

    // private 변수에 접근할려고 getMethod 라는걸 만들어줘야해요
    int getPrice() {
        return price;
    }
}
public class Test1 {
	public static void main(String[] args) {
		System.out.println(Size.LARGE.getPrice());
	}
}

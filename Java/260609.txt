package itinfo_2605_4to7;

import java.util.ArrayList;
import java.util.List;
import java.util.Random;

/* 사용자가 만든 자료형
 * 꼭 단어를 쓰는것같이 가독성이 높아지고, 또 정한값만 가질수 있어서
 * 통일성이 있음 */
enum OrderStatus{
	READY, SHIPPED, CANCLE
}
public class Test1 {
	public static void main(String[] args) {
		// 단순 숫자로 상태를 표현하면 햇갈림
		 int nstatus=3;
		 // enum으로 상태를 표현하면 가동성이 좋음
		 OrderStatus status=OrderStatus.READY;
		 if(status==OrderStatus.READY) {
			 System.out.println("주문 완료");
		 }
		 else if(status==OrderStatus.SHIPPED) {
			 System.out.println("배송 완료");
		 }
		 if(nstatus==0) {
			 System.out.println("준비중");
		 }
		 if(nstatus==5) {
			 System.out.println("취소");
		 }
	}
}

package itinfo_2605_4to7;

import java.util.ArrayList;
import java.util.List;
import java.util.Random;

/* 이 예제는 ENUM을 이악물고 숫자로 보고싶다 할때 쓰는 방법 */
enum Level {
    LOW(1), // ENUM값에 숫자값을 정해줌
    MIDDLE(2),
    HIGH(3);

    int score; // 숫자를 담을 변수

 // ENUM 옆에 있는 값을 스스로 가져가서 score에 셋팅하는 생성자
    Level(int score) { 
        this.score = score;
    }
}
public class Test1 {
	public static void main(String[] args) {
		/* 그럼 사용할때 자동으로 생성자에 숫자 넘겨서 score에 셋팅함 */
		System.out.println(Level.LOW.score);
        System.out.println(Level.HIGH.score);
	}
}

    LARGE(20);

	public static void main(String[] args) {
		System.out.println(Size.LARGE.getPrice());
	}

package itinfo_2605_4to7;

import java.util.ArrayList;
import java.util.List;
import java.util.Random;

/* Thread는 자바 만든사람이 만들어놓은 클래스에요
 * Thread를 상속받으면 run method를 override 할수있게 되요
 * run method 안에다가 내가 하고싶은거 적으면 되요 */
class Worker extends Thread {
	@Override
    public void run() {
        System.out.println("worker");
    }
}
public class Test1 {
	public static void main(String[] args) {
		Worker w = new Worker();
		/* override한 run method를 thread로 동시에 처리하고 싶으면 
		 * start method로 실행해요*/
        w.start();
	}
}

package itinfo_2605_4to7;

import java.util.ArrayList;
import java.util.List;
import java.util.Random;

/* extends Thread 하면 다른클래스를 상속 못받아요
 * 그래서 multi threading 구현할땐 Runnable을 상속 받아요 */
class Task implements Runnable {
	/* 얘도 run을 오바라이드해서 여기에 내가 원하는 기능 적어요 */
	@Override
    public void run() {
        System.out.println("task");
    }
}
public class Test1 {
	public static void main(String[] args) {
		/* 이건 Runnable 상속받은놈을 multithreading 으로
		 * 돌리는 고정된 방법이에요 */
        Task task = new Task();
        Thread t = new Thread(task);
        t.start();
	}
}

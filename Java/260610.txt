package itinfo_2605_4to7;

import java.util.ArrayList;
import java.util.List;
import java.util.Random;

interface Printer {
    void print();
}
public class Test1 {
	public static void main(String[] args) {
		/* 익명 클래스(이름이 없는 클래스)
		 * Printer interface를 상속받고
		 * 클래스 이름은 없이 바로 만들어지는 객체 */
		Printer p = new Printer() {
            public void print() {
                System.out.println("Hello");
            }
        };
        p.print();
	}
}

package itinfo_2605_4to7;

import java.util.ArrayList;
import java.util.List;
import java.util.Random;

interface Calculator {
    int calc(int a, int b);
}
public class Test1 {
	public static void main(String[] args) {
		/* class 만들고, implement 하고
		 * override 하고 new 하는 코드
		 * 다 생략하고, 바로 인터페이스 기반의 객체 만들고
		 * 함수 정의함. 수많은 과정이 생략된거
		 * 매개변수 int a,int b 받고 a+b 결과를 리턴 */
		Calculator c = (a, b) -> a + b;
        System.out.println(c.calc(3, 4));
	}
}

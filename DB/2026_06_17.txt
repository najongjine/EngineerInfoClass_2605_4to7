CREATE TABLE t_user(

);

DROP TABLE t_board;
CREATE TABLE t_board(
id SERIAL PRIMARY KEY,
title VARCHAR NOT NULL,
content VARCHAR,
created_dt TIMESTAMPTZ DEFAULT now(),
user_id INT
);

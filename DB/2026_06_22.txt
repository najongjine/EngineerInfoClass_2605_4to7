CREATE TABLE t_user(
id SERIAL PRIMARY KEY,
username VARCHAR,
name VARCHAR,
email VARCHAR,
created_dt TIMESTAMPTZ DEFAULT now()
);

CREATE TABLE t_board(
id SERIAL PRIMARY KEY,
title VARCHAR,
content TEXT,
created_dt TIMESTAMPTZ DEFAULT now(),
user_id INT NOT NULL,
view_cnt INT DEFAULT 0,
like_cnt INT DEFAULT 0,

CONSTRAINT fk_tboard_userid
FOREIGN KEY (user_id)
REFERENCES t_user(id)
ON DELETE CASCADE
ON UPDATE CASCADE
);

CREATE TABLE t_comment(
id SERIAL PRIMARY KEY,
cmt VARCHAR,
board_id INT NOT NULL,

CONSTRAINT fk_comment_boardid
FOREIGN KEY (board_id)
REFERENCES t_board(id)
ON DELETE CASCADE
ON UPDATE CASCADE
);
